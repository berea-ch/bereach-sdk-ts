# TagsUnion

Add/remove/set tags


## Supported Types

### `operations.Tags1`

```typescript
const value: operations.Tags1 = {
  add: [
    "<value 1>",
  ],
};
```

### `operations.Tags2`

```typescript
const value: operations.Tags2 = {
  remove: [
    "<value 1>",
    "<value 2>",
  ],
};
```

### `operations.Tags3`

```typescript
const value: operations.Tags3 = {
  set: [],
};
```
